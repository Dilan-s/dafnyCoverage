 306          } else {
 307            foreach (var c in s) {
 308              yield return c;
 309            }
 310          }
 311        }
 312      } else {
 313        foreach (var s in TokensWithEscapes(p, false)) {
 314          switch (s) {

 316            case @"\""": yield return '"'; break;
 317            case @"\\": yield return '\\'; break;
 318            case @"\0": yield return '\0'; break;
 319            case @"\n": yield return '\n'; break;

 323              yield return Convert.ToInt32(s[2..], 16);
 324              break;

 333                yield return c;

 337        }

 341    /// <summary>
 342    /// Enumerates the sequence of regular characters and escape sequences in the given well-parsed string.
 343    /// For example, "ab\tuv\u12345" may be broken up as ["a", "b", "\t", "u", "v", "\u1234", "5"].
 344    /// Consecutive non-escaped characters may or may not be enumerated as a single string.

 346    public static IEnumerable<string> TokensWithEscapes(string p, bool isVerbatimString) {
 347      Contract.Requires(p != null);
 348      if (isVerbatimString) {
 349        var skipNext = false;
 350        foreach (var ch in p) {
 351          if (skipNext) {
 352            skipNext = false;
 353          } else {
 354            if (ch == '"') {
 355              skipNext = true;
 356              yield return "\"";
 357            } else {
 358              yield return ch.ToString();
 359            }
 360          }
 361        }
 362      } else {
 363        var i = 0;
 364        while (i < p.Length) {
 365          if (p[i] == '\\') {
 366            switch (p[i + 1]) {
 367              case 'u':
 368                yield return p[i..(i + 6)];
 369                i += 6;
 370                break;
 371              case 'U':
 372                var escapeEnd = p.IndexOf('}', i + 2) + 1;
 373                yield return p[i..escapeEnd];
 374                i = escapeEnd;
 375                continue;
 376              default:
 377                yield return p[i..(i + 2)];
 378                i += 2;
 379                break;
 380            }
 381          } else if (char.IsHighSurrogate(p[i])) {
 382            yield return p[i..(i + 2)];
 383            i += 2;
 384          } else {
 385            yield return p[i].ToString();
 386            i++;
 387          }
 388        }
 389      }
 390    }
