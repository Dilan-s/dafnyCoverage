 7/// <summary>
 8/// AutoContracts is an experimental feature that will fill much of the dynamic-frames boilerplate
 9/// into a class.  From the user's perspective, what needs to be done is simply:
 10///  - mark the class with {:autocontracts}
 11///  - declare a function (or predicate) called Valid()
 12///
 13/// AutoContracts will then:
 14///
 15/// Declare, unless there already exist members with these names:
 16///    ghost var Repr: set(object)
 17///    predicate Valid()
 18///
 19/// For function/predicate Valid(), insert:
 20///    reads this, Repr
 21///    ensures Valid() ==> this in Repr
 22/// Into body of Valid(), insert (at the beginning of the body):
 23///    this in Repr && null !in Repr
 24/// and also insert, for every array-valued field A declared in the class:
 25///    (A != null ==> A in Repr) &&
 26/// and for every field F of a class type T where T has a field called Repr, also insert:
 27///    (F != null ==> F in Repr && F.Repr SUBSET Repr && this !in Repr && F.Valid())
 28/// Except, if A or F is declared with {:autocontracts false}, then the implication will not
 29/// be added.
 30///
 31/// For every constructor, add:
 32///    ensures Valid() && fresh(Repr)
 33/// At the end of the body of the constructor, add:
 34///    Repr := {this};
 35///    if (A != null) { Repr := Repr + {A}; }
 36///    if (F != null) { Repr := Repr + {F} + F.Repr; }
 37///
 38/// In all the following cases, no "modifies" clause or "reads" clause is added if the user
 39/// has given one.
 40///
 41/// For every non-static non-ghost method that is not a "simple query method",
 42/// add:
 43///    requires Valid()
 44///    modifies Repr
 45///    ensures Valid() && fresh(Repr - old(Repr))

 47///    if (A != null && !(A in Repr)) { Repr := Repr + {A}; }
