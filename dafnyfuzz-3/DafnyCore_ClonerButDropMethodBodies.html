 743    public override ModuleDefinition CloneModuleDefinition(ModuleDefinition m, Name name) {
 744      var basem = base.CloneModuleDefinition(m, name);
 745
 746      //Merge signatures for imports which point to the same module
 747      //This makes the consistency check understand that the same element
 748      //may be referred to via different qualifications.
 749      var sigmap = new Dictionary<ModuleDefinition, ModuleSignature>();
 750      var declmap = new Dictionary<ModuleDefinition, List<AliasModuleDecl>>();
 751      var vismap = new Dictionary<ModuleDefinition, VisibilityScope>();
 752
 753      foreach (var top in basem.TopLevelDecls) {
 754        var import = reverseMap[top] as AliasModuleDecl;
 755        if (import == null) {
 756          continue;
 757        }
 758
 759        var def = import.Signature.ModuleDef;
 760        if (def == null) {
 761          continue;
 762        }
 763
 764        if (!declmap.ContainsKey(def)) {
 765          declmap.Add(def, new List<AliasModuleDecl>());
 766          sigmap.Add(def, new ModuleSignature());
 767          vismap.Add(def, new VisibilityScope());
 768        }
